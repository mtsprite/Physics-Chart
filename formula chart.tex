\begin{multicols}{3}[\section*{\underline {Linear:}}]

%velocity
$v = \frac{\Delta x}{\Delta t}$\\
%acceleration
$a = \frac{\Delta v}{\Delta t}$\\
\\
%kinematics
$v_f = v_i + at$\\
$x_f = x_i + v_it + \frac{1}{2}at^2$\\
$v^2 = v_i^2 + 2a(x - x_i)$\\
$v = \frac{1}{2}(v + v_i)$\\
\\
%Force Net
$\Sigma F_{net} = ma$\\
$F_{F} = \mu F_{N}$\\
$F_{N} = mg\cos\theta$\\
\\
%work
$W = F_{net}\Delta x\cos\theta$\\
$ F_{net}\Delta x\cos\theta = \Delta E$\\
$W = \frac{1}{2}m{v_f}^2 - \frac{1}{2}m{v_i}^2$\\
$F = -k\Delta x$\\
\\
%power
$P = \frac{\Delta W}{\Delta t}$\\
$P = Fv\cos\theta$\\
$P = Fv$\\
\\
%energy
$KE = \frac{1}{2}mv^2$\\
$PE_{grav} = mgh$\\
$PE_{spring} = \frac{1}{2}kx^2$\\
\\
%momentum
$p  = m\Delta v$\\
\\
%impulse
$J = F_{net}\Delta t$\\
$J = F_{net}\Delta t = \Delta p =mv_f - mv_i$
\end{multicols}
\begin{multicols}{3}[\section*{\underline {Rotational:}}]
%period
$T = \frac{1}{f}$\\
$T = \frac{2\pi}{\omega}$\\
%frequency
$f = \frac{1}{T}$\\
%centripital velocity
$v_c = \frac{2\pi r}{T}$\\
%centripital acceleration
$a_c = \frac{\Delta v}{t}$\\
%centripital force
$F_c = m\frac{{v_c}^2}{r}$\\
$F_c = ma$\\
$F_c = m\frac{4{\pi}^2r}{T^2}$\\
\end{multicols}
\begin{multicols}{3}[\section*{\underline {Orbital:}}]
%gravitational constant

%gravitational force
$F_{grav} = -\frac{Gm_1m_2}{r}$\\
%gravitational potential energy
$U_g = -\frac{Gm_1m_2}{r^2}$\\
%gravitational potential
$F_{grav} = \frac{GMm}{r}$\\
\end{multicols}
\begin{multicols}{3}[\section*{\underline {Angular:}}]
%angular acceleration
$\alpha = \frac{\Delta\omega}{t}$\\
%linear acceleration
$a = \alpha r-\omega^2 r$\\
%angular velocity
$\omega = \frac{\Delta\theta}{t}$\\
$\omega = 2 \pi f$\\
%linear velocity
$v = \omega * r$\\
\\
%angular kinematics
$\omega_f = \omega_i + at$\\
$\theta_f = \theta_i + \omega_it + \frac{1}{2}\alpha t^2$\\
${\omega_f}^2 = {\omega_i}^2 + 2\alpha(\theta_f-\theta_i)$\\
$\omega = \frac{1}{2}(\omega_f-\omega_i)$\\
$\theta = \frac{1}{2}(\omega_f-\omega_i)t$\\
\\
%torque
$\tau = F_{net}r\sin\theta$\\
$\tau =  I\alpha$\\
$\tau = \frac{L}{t}$\\
\\
%inertia
$I = \Sigma mr^2$\\
\\
%angular momentum
$L = mrv\sin\theta$\\
$L = rp$\\
$L = I\omega$\\
\\
%rotational kenetic energy
$K = \frac{1}{2}I\omega^2$\\
\\
%rotational power
$P = \tau\omega\cos\theta$\\
$P = \tau\omega$\\
\\
%rotational work
$W = \tau\Delta\theta$\\
\end{multicols}
